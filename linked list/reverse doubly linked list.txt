#include <bits/stdc++.h>
using namespace std;

struct Node{
	   int data;
	   Node* next;
	   Node* prev;
	   Node(int x){
	       data=x;
	       next=NULL;
	       prev=NULL;
	   }
};

Node* reverseList(Node* head){
    Node* curr=head;
    Node* prev=NULL;
    Node* next=NULL;
    while(curr!=NULL){
        next=curr->next;
        curr->next=prev;
        curr->prev=next;
        
        prev=curr;
        curr=next;
        
    }
    return prev;
}


void display(Node* head){
    Node* curr=head;
    while(curr!=NULL){
        cout<<curr->data<<endl;
        curr=curr->next;
    }
}


int main() {
	// your code goes here
	Node* head=new Node(10);
	Node* temp1=new Node(20);
	Node* temp2=new Node(30);
	Node* temp3=new Node(40);
	
	head->next=temp1;
	
	temp1->next=temp2;
	temp1->prev=head;
	
	temp2->next=temp3;
	temp2->prev=temp1;
	
	temp3->prev=temp2;
	
	head=reverseList(head);
	
	display(head);
	
}
