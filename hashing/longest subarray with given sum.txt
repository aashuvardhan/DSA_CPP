#include <bits/stdc++.h>
using namespace std;

int main() {
	// your code goes here
	vector<int> v={8,3,1,5,-6,6,2,2};
	int sum=4;
	
	unordered_map<int,int> umap;
	int prefix_sum=0;
	int res=0;
	
	for(int i=0;i<v.size();i++){
	    prefix_sum+=v[i];
	    
	    if(prefix_sum==sum){
	        res=i+1;
	    }
	    if(umap.find(prefix_sum)==umap.end()){
	        umap[prefix_sum]=i;
	    }
	    if(umap.find(prefix_sum-sum)!=umap.end()){
	        res=max(res,i-umap[prefix_sum-sum]);
	    }
	    
	}
	cout<<res;
	

}
